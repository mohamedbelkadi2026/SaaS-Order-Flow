I want a third option that's the simplest possible: user pastes a public Google Sheet URL, backend polls it via the public CSV export endpoint.How it worksUser flow:
  1. User opens their Google Sheet
  2. Share → "Anyone with the link" → Viewer
  3. Copy URL
  4. Paste in TajerGrow → Click "Connecter"
  5. Backend validates URL by fetching one row → confirms format
  6. Done. Cron polls every 5 minutes.

// Accepts: https://docs.google.com/spreadsheets/d/{ID}/edit#gid=0
//          https://docs.google.com/spreadsheets/d/{ID}/edit
//          https://docs.google.com/spreadsheets/d/{ID}
function extractSheetId(url: string): string | null {
  const m = url.match(/\/spreadsheets\/d\/([a-zA-Z0-9_-]+)/);
  return m ? m[1] : null;
}

// Fetch tabs metadata from a public Google Sheet using the OG document HTML
async function fetchPublicSheetTabs(sheetId: string): Promise<Array<{ gid: string; title: string }>> {
  // Public CSV-export URLs need the gid; we get the gid list by parsing the
  // sheet's HTML viewer page (no API key needed for public sheets).
  const htmlUrl = `https://docs.google.com/spreadsheets/d/${sheetId}/htmlview`;
  const resp = await fetch(htmlUrl, { redirect: 'follow' });
  if (!resp.ok) throw new Error(`Sheet not accessible — verify it's set to 'Anyone with the link can view'`);
  const html = await resp.text();

  // The HTML embeds a JS object with sheet titles + gids. Look for patterns like:
  //   {"name":"cofer","gid":1234567}
  // or in newer versions:
  //   "sheets":[{"properties":{"sheetId":0,"title":"Sheet1"}}]
  const tabs: Array<{ gid: string; title: string }> = [];

  // Pattern 1: legacy htmlview
  const re1 = /(?:name|title)["']?\s*:\s*["']([^"']+)["']\s*,\s*(?:sheetId|gid)["']?\s*:\s*["']?(\d+)/g;
  let match;
  while ((match = re1.exec(html)) !== null) {
    if (!tabs.find(t => t.gid === match![2])) {
      tabs.push({ title: match[1], gid: match[2] });
    }
  }

  // Fallback: parse anchor tags <a id="sheet-button-X">TabName</a> + extract gids from links
  if (tabs.length === 0) {
    const linkRe = /<a[^>]+id=["']sheet-button-(\d+)["'][^>]*>([^<]+)<\/a>/g;
    while ((match = linkRe.exec(html)) !== null) {
      tabs.push({ gid: match[1], title: match[2].trim() });
    }
  }

  // Last resort: just gid=0
  if (tabs.length === 0) tabs.push({ gid: '0', title: 'Sheet1' });

  return tabs;
}

// POST /api/integrations/google-sheets/connect-url
app.post('/api/integrations/google-sheets/connect-url', requireAuth, async (req, res) => {
  const { url, magasinId } = req.body as { url: string; magasinId?: number };
  if (!url) return res.status(400).json({ error: 'URL manquante' });

  const sheetId = extractSheetId(url);
  if (!sheetId) {
    return res.status(400).json({
      error: "URL invalide. Collez l'URL complète de votre Google Sheet (commençant par https://docs.google.com/spreadsheets/d/...)",
    });
  }

  // Validation: try to fetch tabs from the sheet
  let tabs: Array<{ gid: string; title: string }>;
  try {
    tabs = await fetchPublicSheetTabs(sheetId);
  } catch (err: any) {
    return res.status(400).json({
      error: "Impossible d'accéder à votre Google Sheet. Vérifiez qu'il est bien partagé avec 'Tout le monde avec le lien' (Lecteur).",
      detail: err.message,
    });
  }

  // Probe one tab to confirm CSV export works
  const probeUrl = `https://docs.google.com/spreadsheets/d/${sheetId}/export?format=csv&gid=${tabs[0].gid}`;
  const probeResp = await fetch(probeUrl);
  if (!probeResp.ok) {
    return res.status(400).json({
      error: "Le sheet n'est pas réellement public. Activez 'Tout le monde avec le lien — Lecteur' et réessayez.",
    });
  }

  // Persist
  const storeId = req.user!.storeId!;
  const existing = await db.select().from(storeIntegrations)
    .where(and(eq(storeIntegrations.storeId, storeId), eq(storeIntegrations.provider, 'gsheets')))
    .limit(1);

  if (existing.length > 0) {
    await db.update(storeIntegrations)
      .set({
        gsheetUrl: url,
        gsheetId: sheetId,
        gsheetTabs: tabs,
        gsheetSyncState: {},
        magasinId: magasinId || null,
        status: 'active',
      } as any)
      .where(eq(storeIntegrations.id, existing[0].id));
  } else {
    await db.insert(storeIntegrations).values({
      storeId,
      magasinId: magasinId || null,
      provider: 'gsheets',
      apiKey: '',
      gsheetUrl: url,
      gsheetId: sheetId,
      gsheetTabs: tabs,
      gsheetSyncState: {},
      status: 'active',
    } as any);
  }

  res.json({
    success: true,
    sheetId,
    tabsCount: tabs.length,
    tabs: tabs.map(t => t.title),
  });
});

// CSV parser — basic but handles quoted fields with commas inside
function parseCsv(text: string): string[][] {
  const rows: string[][] = [];
  let row: string[] = [];
  let cell = '';
  let inQuotes = false;
  for (let i = 0; i < text.length; i++) {
    const c = text[i];
    if (inQuotes) {
      if (c === '"' && text[i + 1] === '"') { cell += '"'; i++; }
      else if (c === '"') inQuotes = false;
      else cell += c;
    } else {
      if (c === '"') inQuotes = true;
      else if (c === ',') { row.push(cell); cell = ''; }
      else if (c === '\r') { /* skip */ }
      else if (c === '\n') { row.push(cell); rows.push(row); row = []; cell = ''; }
      else cell += c;
    }
  }
  if (cell !== '' || row.length > 0) { row.push(cell); rows.push(row); }
  return rows;
}

// Positional column mapping (legacy sheets without headers)
const POSITIONAL = {
  name: 0, phone: 1, city: 2, address: 3,
  product: 4, price: 5, quantity: 6,
  utmCampaign: 10, utmSource: 11, productId: 12,
};

export async function syncAllPublicSheets() {
  const conns = await db.select().from(storeIntegrations)
    .where(and(
      eq(storeIntegrations.provider, 'gsheets'),
      eq(storeIntegrations.status, 'active'),
      sql`${(storeIntegrations as any).gsheetUrl} IS NOT NULL`,
    ));

  for (const conn of conns) {
    try {
      await syncOne(conn as any);
    } catch (err: any) {
      console.error(`[GSHEETS-PUBLIC] Store ${conn.storeId} error:`, err.message);
    }
  }
}

async function syncOne(conn: any) {
  const sheetId = conn.gsheetId;
  const tabs = conn.gsheetTabs || [];
  const state = conn.gsheetSyncState || {};

  for (const tab of tabs) {
    const lastRow = state[`tab_${tab.gid}`] || 0;

    const csvUrl = `https://docs.google.com/spreadsheets/d/${sheetId}/export?format=csv&gid=${tab.gid}`;
    const resp = await fetch(csvUrl);
    if (!resp.ok) {
      console.warn(`[GSHEETS-PUBLIC] Tab ${tab.title} fetch failed: ${resp.status}`);
      continue;
    }
    const text = await resp.text();
    const rows = parseCsv(text);

    // Heuristic: detect header row. If row 0 contains text-only cells with
    // recognizable words ('nom', 'phone', 'name', etc.), skip it.
    let dataStart = 0;
    if (rows.length > 0) {
      const first = rows[0].map(c => c.toLowerCase().trim()).join(' ');
      if (/\b(nom|name|phone|telephone|tel|address|city|ville|product|price|prix|fullname)\b/.test(first)) {
        dataStart = 1;
      }
    }

    // Process rows newer than last_sync
    const newRows = rows.slice(Math.max(dataStart, lastRow));

    for (let i = 0; i < newRows.length; i++) {
      const row = newRows[i];
      const rowIndex = Math.max(dataStart, lastRow) + i;

      const name  = (row[POSITIONAL.name]  || '').trim();
      const phone = (row[POSITIONAL.phone] || '').trim();
      if (!name && !phone) continue;
      if (!phone) continue;  // skip rows without phone

      const ref = `GSP-${sheetId.slice(0, 6)}-${tab.gid}-R${rowIndex + 1}`;

      // Idempotency: skip if order with this ref already exists
      const existing = await db.select({ id: orders.id }).from(orders)
        .where(eq(orders.orderNumber, ref))
        .limit(1);
      if (existing.length > 0) continue;

      const totalPriceCents = Math.round(
        parseFloat(String(row[POSITIONAL.price] || '0').replace(',', '.')) * 100
      ) || 0;
      const product = (row[POSITIONAL.product] || '').trim() || tab.title;
      const qty     = parseInt(row[POSITIONAL.quantity] || '1', 10) || 1;

      try {
        await storage.createOrder({
          storeId:         conn.storeId,
          magasinId:       conn.magasinId,
          orderNumber:     ref,
          customerName:    name,
          customerPhone:   phone,
          customerAddress: (row[POSITIONAL.address] || '').trim(),
          customerCity:    (row[POSITIONAL.city] || '').trim(),
          rawProductName:  product,
          status:          'nouveau',
          totalPrice:      totalPriceCents,
          productCost:     0,
          shippingCost:    0,
          adSpend:         0,
          source:          'gsheets',
          comment:         null,
          utmSource:       (row[POSITIONAL.utmSource] || '').trim() || null,
          utmCampaign:     (row[POSITIONAL.utmCampaign] || '').trim() || null,
        } as any, [{
          rawProductName: product,
          quantity: qty,
          price: totalPriceCents,
        }]);
        console.log(`[GSHEETS-PUBLIC] Created order ref=${ref} from tab=${tab.title} row=${rowIndex + 1}`);
      } catch (err: any) {
        console.error(`[GSHEETS-PUBLIC] Order creation failed for ref=${ref}: ${err.message}`);
      }
    }

    state[`tab_${tab.gid}`] = rows.length;
  }

  await db.update(storeIntegrations)
    .set({
      gsheetSyncState: state,
      lastSyncAt: new Date(),
    } as any)
    .where(eq(storeIntegrations.id, conn.id));
