🛡️ P0-2 — Fix email enumeration on login
Why
Currently login distinguishes "Email incorrect" vs "Mot de passe incorrect" → attackers can enumerate which emails are registered → targeted phishing.
File: server/auth.ts
Around line 100-110, replace:
tsasync (email, password, done) => {
  try {
    const user = await storage.getUserByEmail(email);
    if (!user) return done(null, false, { message: "Email incorrect" });
    if (!user.password) return done(null, false, { message: "Mot de passe non configuré" });
    const valid = await comparePasswords(password, user.password);
    if (!valid) return done(null, false, { message: "Mot de passe incorrect" });
    if (user.isActive === 0 && !user.isSuperAdmin) return done(null, false, { message: "Votre compte est suspendu. Veuillez contacter l'administration." });
    return done(null, user);
  } catch (err) {
    return done(err);
  }
}
With:
tsasync (email, password, done) => {
  // Generic message — never tell attackers which step failed.
  // Constant-time comparison even when user doesn't exist (timing attack defense).
  const GENERIC_FAIL = { message: "Email ou mot de passe incorrect" };
  const DUMMY_HASH = "0".repeat(128) + "." + "0".repeat(32); // valid format, won't match anything

  try {
    const user = await storage.getUserByEmail(email);

    // Always run the password comparison, even for non-existent users.
    // Prevents timing attacks that could detect whether the email exists.
    const passwordToCheck = user?.password || DUMMY_HASH;
    const valid = await comparePasswords(password, passwordToCheck).catch(() => false);

    if (!user || !user.password || !valid) {
      return done(null, false, GENERIC_FAIL);
    }

    if (user.isActive === 0 && !user.isSuperAdmin) {
      return done(null, false, { message: "Votre compte est suspendu. Veuillez contacter l'administration." });
    }

    return done(null, user);
  } catch (err) {
    return done(err);
  }
}
Acceptance criteria

✅ Login with wrong email → "Email ou mot de passe incorrect"
✅ Login with right email + wrong password → same message
✅ Suspended account → distinct "Votre compte est suspendu" (this one's OK to differentiate, since attacker already had the right password)
✅ Login response time consistent (~80-150ms) regardless of email existence

🛡️ P0-3 — Add session regeneration on login
Why
Without regenerating session ID on login, attackers can fixate a session before user logs in, then hijack it after.
File: server/routes.ts (or wherever the login endpoint lives)
Find the login endpoint (search passport.authenticate('local'). Replace the success handler to regenerate:
tsapp.post("/api/auth/login", (req, res, next) => {
  passport.authenticate("local", (err: any, user: any, info: any) => {
    if (err) return next(err);
    if (!user) return res.status(401).json({ message: info?.message || "Échec de connexion" });

    // ── SECURITY: regenerate session ID after successful auth ─────────────
    // Prevents session fixation attacks. The pre-login session is destroyed,
    // a fresh one is created, and the user is logged into the fresh one.
    req.session.regenerate((regenErr) => {
      if (regenErr) return next(regenErr);
      req.login(user, (loginErr) => {
        if (loginErr) return next(loginErr);
        // Save the session before sending response (avoid race conditions)
        req.session.save((saveErr) => {
          if (saveErr) return next(saveErr);
          res.json({ user: { id: user.id, email: user.email, role: user.role, username: user.username } });
        });
      });
    });
  })(req, res, next);
});
Also: on logout, destroy the session (not just remove user):
tsapp.post("/api/auth/logout", (req, res) => {
  req.logout((err) => {
    if (err) return res.status(500).json({ message: err.message });
    req.session.destroy(() => {
      res.clearCookie('connect.sid');
      res.json({ ok: true });
    });
  });
});
Acceptance criteria

✅ Inspect cookies before login: connect.sid = X
✅ Login successfully → cookies show connect.sid = Y (different value)
✅ Logout → cookie cleared from browser
✅ User stays logged in across page refreshes after login (no regression)

🛡️ P0-6 — Whitelist all db.insert / db.update payloads (mass assignment defense)
Why
The biggest privilege-escalation vulnerability in many SaaS. Without explicit field whitelisting, an attacker passes extra fields in JSON body and they get written to the database.
Example:
bash# Update profile → tries to make self super admin
curl -X PUT https://www.tajergrow.com/api/users/me \
  -H "Cookie: connect.sid=..." \
  -d '{"username":"Hassan","isSuperAdmin":1,"role":"admin"}'
If the endpoint does db.update(users).set(req.body) → attacker becomes super admin.
What to do
This is a systematic audit, not a single edit. Search the codebase for risky patterns:
bash# Run these greps to find candidates:
grep -rn "db\.insert.*\.values(req\.body" server/
grep -rn "db\.update.*\.set(req\.body" server/
grep -rn "\.\.\.req\.body" server/
Each match needs to be reviewed and replaced with explicit field destructuring.
Pattern to apply
BEFORE (vulnerable):
tsapp.put("/api/users/:id", requireAuth, async (req, res) => {
  await db.update(users).set(req.body).where(eq(users.id, req.params.id));
  res.json({ ok: true });
});
AFTER (safe):
tsimport { z } from 'zod';

const updateUserSchema = z.object({
  username: z.string().min(2).max(80).optional(),
  phone:    z.string().regex(/^(\+212|0)[5-7][0-9]{8}$/).optional(),
  email:    z.string().email().optional(),
  // NOTE: role, isSuperAdmin, isActive, storeId are NOT in this schema —
  // users can't change those via this endpoint.
});

app.put("/api/users/:id", requireAuth, async (req, res) => {
  // Permission check
  const targetId = Number(req.params.id);
  if (req.user!.id !== targetId && !req.user!.isSuperAdmin) {
    return res.status(403).json({ message: "Accès refusé" });
  }

  // Validate input
  const parsed = updateUserSchema.safeParse(req.body);
  if (!parsed.success) {
    return res.status(400).json({ errors: parsed.error.flatten() });
  }

  // Update with whitelisted fields only
  await db.update(users).set(parsed.data).where(eq(users.id, targetId));
  res.json({ ok: true });
});

Generic helper to make this pattern reusable
Create server/utils/validate.ts:
tsimport type { Request, Response, NextFunction } from 'express';
import type { z } from 'zod';

/**
 * Middleware: validates req.body against a Zod schema.
 * On success, replaces req.body with the parsed (typed, sanitized) value.
 * On failure, returns 400 with detailed error info.
 */
export function validateBody<T extends z.ZodTypeAny>(schema: T) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({
        message: result.error.errors[0]?.message || 'Validation failed',
        errors: result.error.flatten(),
      });
    }
    req.body = result.data;
    next();
  };
}

/**
 * Permission helper: ensures the caller can act on the given store.
 * Throws 403 otherwise.
 */
export function requireStoreOwnership(storeId: number) {
  return async (req: Request, res: Response, next: NextFunction) => {
    if (!req.user) return res.status(401).json({ message: 'Non authentifié' });
    if ((req.user as any).isSuperAdmin) return next();
    if ((req.user as any).storeId === storeId) return next();
    return res.status(403).json({ message: 'Magasin non autorisé' });
  };
}
Use it like:
tsapp.put("/api/users/:id", requireAuth, validateBody(updateUserSchema), async (req, res) => {
  // req.body is now typed & safe
});

✅ Attempting to set isSuperAdmin: 1 via any endpoint → field ignored, user stays non-admin
✅ Attempting to set storeId to another store via order/product creation → field ignored, server forces caller's storeId
✅ All schemas live in shared/schema.ts so frontend can also validate
✅ TypeScript build passes — no implicit any from req.body
