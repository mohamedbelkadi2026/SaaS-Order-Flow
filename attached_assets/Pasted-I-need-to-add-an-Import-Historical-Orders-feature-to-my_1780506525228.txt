I need to add an "Import Historical Orders" feature to my TajerGrow platform. New clients come with their existing data in Google Sheets/Excel: confirmed orders, delivered orders (with tracking), pas de réponse orders, etc. They need to import this history INTO the platform WITHOUT losing the real status or tracking numbers, so they can start from where they were instead of from scratch.

THE GOAL
═══════════════════════════════════════════════════════════════════════
1. A new client connects their account.
2. They go to "Importer Historique" page.
3. They upload an Excel/CSV file (exported from Google Sheets) OR paste a Google Sheets URL.
4. The system:
   - Maps columns (name, phone, address, city, product, price, status, tracking, date, etc.)
   - Lets the user PREVIEW + map columns + select default magasin/store
   - Bulk creates orders in their store with:
     * Correct status (Confirmé / Pas de réponse / Livré / Annulé / etc.)
     * Tracking number preserved (if delivered/shipped)
     * Carrier name preserved (if known)
     * Original creation date preserved
     * Products + quantities
   - Skips duplicates (same phone + product within X days, or same tracking number)

  app.post("/api/orders/import-bulk", requireAuth, async (req: any, res: any) => {
    const storeId = req.user!.storeId!;
    try {
      const {
        rows,                  // array of mapped order objects
        magasinId,             // user-picked magasin
        defaultStatus,         // fallback status if a row has none
        defaultSource,         // e.g. "import"
        skipDuplicates,        // boolean
        duplicateStrategy,     // "phone+product" | "tracking" | "both"
      } = req.body;

      if (!Array.isArray(rows) || rows.length === 0) {
        return res.status(400).json({ message: "Aucune ligne à importer" });
      }
      if (rows.length > 5000) {
        return res.status(400).json({ message: "Maximum 5000 commandes par import. Divisez votre fichier." });
      }

      // Pre-fetch existing orders for duplicate detection (last 90 days, this store)
      const cutoff = new Date(Date.now() - 90 * 86400 * 1000);
      const existingOrders = await storage.getOrdersSince(storeId, cutoff);

      const phoneProductSet = new Set<string>();
      const trackingSet = new Set<string>();
      for (const o of existingOrders) {
        if (o.customerPhone) {
          // We don't know product here cheaply; use phone+date approximation
          phoneProductSet.add(`${(o.customerPhone || "").replace(/\D/g, "")}|${o.customerName?.toLowerCase().trim() || ""}`);
        }
        if (o.trackingNumber) trackingSet.add(o.trackingNumber);
      }

      const results = { created: 0, skipped: 0, errors: [] as any[] };

      for (let i = 0; i < rows.length; i++) {
        const row = rows[i];
        try {
          const phone = String(row.customerPhone || "").replace(/\D/g, "");
          const name = String(row.customerName || "").trim();
          const tracking = String(row.trackingNumber || "").trim();

          // Duplicate detection
          const dupKey = `${phone}|${name.toLowerCase()}`;
          const isDupByPhoneName = skipDuplicates && phoneProductSet.has(dupKey);
          const isDupByTracking = skipDuplicates && tracking && trackingSet.has(tracking);
          if (isDupByPhoneName || isDupByTracking) {
            results.skipped += 1;
            continue;
          }

          // Map status — accept multiple variants, normalize to platform statuses
          const rawStatus = String(row.status || "").toLowerCase().trim();
          const status = mapImportedStatus(rawStatus, defaultStatus || "Confirmé");

          // Parse date (handle multiple formats: ISO, dd/mm/yyyy, etc.)
          const createdAt = parseFlexibleDate(row.createdAt) || new Date();

          // Auto-create or find product
          let productId: number | null = null;
          let unitPrice = Number(row.price) || 0;
          let quantity = Number(row.quantity) || 1;
          if (row.productName) {
            const product = await storage.getOrCreateProductByName(storeId, {
              name: String(row.productName).trim(),
              sku: row.productSku ? String(row.productSku).trim() : null,
              price: unitPrice,
            });
            productId = product?.id || null;
            if (product?.price && !unitPrice) unitPrice = product.price;
          }

          // Create the order
          const order = await storage.createOrder({
            storeId,
            magasinId: magasinId || null,
            customerName: name || "Client importé",
            customerPhone: phone,
            customerAddress: String(row.customerAddress || "").trim(),
            customerCity: String(row.customerCity || "").trim(),
            status,
            trackingNumber: tracking || null,
            carrierProvider: row.carrierProvider ? String(row.carrierProvider).toLowerCase().trim() : null,
            totalPrice: Math.round((unitPrice * quantity) * 100) / 100,  // adjust if stored as integer cents
            notes: row.notes ? String(row.notes).trim() : null,
            source: defaultSource || "import",
            createdAt,
            updatedAt: createdAt,
          } as any);

          // Add item
          if (productId && order?.id) {
            await storage.createOrderItem({
              orderId: order.id,
              productId,
              quantity,
              price: unitPrice,
            } as any);
          }

          // If delivered, set shippedAt + deliveredAt to the row date
          if (status === "Livré" && order?.id) {
            await storage.updateOrderTimestamps(order.id, {
              shippedAt: createdAt,
              deliveredAt: createdAt,
            });
          }

          // Mark as processed for subsequent duplicate detection in same batch
          phoneProductSet.add(dupKey);
          if (tracking) trackingSet.add(tracking);
          results.created += 1;
        } catch (err: any) {
          results.errors.push({ row: i + 1, message: err.message || "Erreur ligne" });
        }
      }

      res.json(results);
    } catch (err: any) {
      console.error("[ImportBulk] error:", err);
      res.status(500).json({ message: err.message || "Erreur" });
    }
  });

  // Map user-supplied status strings to platform statuses
  function mapImportedStatus(raw: string, fallback: string): string {
    const s = (raw || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").trim();
    if (!s) return fallback;

    // Delivered
    if (["livre", "livree", "delivered", "delivere", "completed", "done", "ok livre"].some(v => s === v || s.includes(v))) return "Livré";
    // Confirmed
    if (["confirme", "confirmed", "confirm", "ok", "valid", "valide", "oui"].some(v => s === v || s.includes(v))) return "Confirmé";
    // No answer
    if (["pas de reponse", "no answer", "ne repond pas", "no reply", "nrp", "no answer yet"].some(v => s === v || s.includes(v))) return "Pas de réponse";
    // Cancelled / refused
    if (["annule", "cancel", "cancelled", "refuse", "refused", "no", "non"].some(v => s === v || s.includes(v))) return "Annulé";
    // Returned
    if (["retour", "return", "returned", "rendu"].some(v => s === v || s.includes(v))) return "Retour";
    // Shipped / in transit
    if (["expedie", "shipped", "in transit", "en cours", "en route"].some(v => s === v || s.includes(v))) return "Expédié";
    // Reported / postponed
    if (["reporte", "report", "postponed", "later"].some(v => s === v || s.includes(v))) return "Reporté";

    return fallback;
  }

  function parseFlexibleDate(input: any): Date | null {
    if (!input) return null;
    if (input instanceof Date) return isNaN(input.getTime()) ? null : input;
    const s = String(input).trim();
    // ISO
    const iso = new Date(s);
    if (!isNaN(iso.getTime())) return iso;
    // dd/mm/yyyy or dd-mm-yyyy
    const m = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{2,4})/);
    if (m) {
      const dd = parseInt(m[1]), mm = parseInt(m[2]) - 1;
      let yy = parseInt(m[3]);
      if (yy < 100) yy += 2000;
      const d = new Date(yy, mm, dd);
      if (!isNaN(d.getTime())) return d;
    }
    return null;
  }

  async function importInChunks(allRows: any[], chunkSize = 200) {
    const totals = { created: 0, skipped: 0, errors: [] as any[] };
    for (let i = 0; i < allRows.length; i += chunkSize) {
      const chunk = allRows.slice(i, i + chunkSize);
      const resp = await fetch("/api/orders/import-bulk", {
        method: "POST",
        credentials: "include",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          rows: chunk,
          magasinId: selectedMagasinId,
          defaultStatus,
          defaultSource: "import",
          skipDuplicates: true,
          duplicateStrategy: "phone+name",
        }),
      });
      const r = await resp.json();
      totals.created += r.created || 0;
      totals.skipped += r.skipped || 0;
      totals.errors.push(...(r.errors || []));
      setProgress(Math.min(100, Math.round(((i + chunkSize) / allRows.length) * 100)));
    }
    return totals;
  }
