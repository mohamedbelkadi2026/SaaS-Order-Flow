  try {
    const stored = localStorage.getItem('tajergrow_all_columns');
    if (stored) {
      const parsed: string[] = JSON.parse(stored);
      // Merge: keep user's existing order + visibility, then append any
      // new columns that have been added to DEFAULT_VISIBLE since they
      // last customized. This means future new columns auto-appear once
      // for every user without losing their personalization.
      const merged = [...parsed];
      for (const col of DEFAULT_VISIBLE) {
        if (!merged.includes(col)) merged.push(col);
      }
      // Also drop any keys that no longer exist in ALL_COLUMNS (cleanup)
      const validKeys = new Set(ALL_COLUMNS.map(c => c.key));
      return merged.filter(k => validKeys.has(k));
    }
  } catch {}
  return DEFAULT_VISIBLE;
}This is the standard pattern for forward-compatible column persistence. After this change:
Existing users keep their custom column order and visibility
The new boutique column appears automatically (appended to the end of their list)
Any future new columns will also appear automatically without manual reset
Removed columns are silently filtered out
