CHANGE 4 — Fix orders already wrongly marked as delivered:
In server/routes.ts, add a one-time fix endpoint to correct wrong statuses:
typescriptapp.post('/api/admin/fix-wrong-delivered', requireAuth, async (req, res) => {
  const storeId = req.user!.storeId!;
  const orders = await storage.getOrdersByStore(storeId);
  let fixed = 0;
  for (const order of orders) {
    const wrongStatuses = ['Confirmé par livreur', 'Confirmé par livreur *', 'En cours de livraison', 'Rappel en cours', 'Rappel en cours *'];
    if (order.status === 'delivered' && wrongStatuses.includes((order as any).commentStatus || '')) {
      await storage.updateOrderStatus(order.id, 'in_progress');
      fixed++;
    }
  }
  res.json({ fixed });
});
