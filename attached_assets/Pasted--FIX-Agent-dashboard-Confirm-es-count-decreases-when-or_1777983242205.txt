  'expédié', 'expedie', 'shipped',
  'attente de ramassage', 'attente ramassage',
  'in_progress', 'inprogress', 'en cours', 'transit',

  'expédié', 'expedie', 'shipped',
  'attente de ramassage', 'attente ramassage',
  'in_progress', 'inprogress', 'en cours', 'transit',
]);
const REFUSED_STATUSES = new Set(['refused', 'refusé', 'refusée', 'refusee', 'retourné']);
const CANCELLED_STATUSES_PREFIX = ['annulé', 'annule'];

const buckets = {
  // Cumulative: every order that ever reached confirmation, no matter where it is now
  confirme:    0,
  // Subsets of cumulative — for breakdown
  delivered:   0,
  en_cours:    0,
  refused:     0,
  // Negative outcomes (never reached confirmation OR explicitly cancelled)
  cancelled:   0,
};

for (const o of agentOrders) {
  const s = (o.status ?? '').toLowerCase().trim();

  // Cumulative confirmation: counted ONCE per order, no matter the current stage
  if (CONFIRMED_STATUSES.has(s)) {
    buckets.confirme++;
  }

  // Sub-breakdown (these are SUBSETS of confirme — already counted above)
  if (DELIVERED_STATUSES.has(s))  buckets.delivered++;
  if (IN_TRANSIT_STATUSES.has(s)) buckets.en_cours++;
  if (REFUSED_STATUSES.has(s))    buckets.refused++;

  // Cancelled is mutually exclusive with confirmed
  if (CANCELLED_STATUSES_PREFIX.some(p => s.startsWith(p))) buckets.cancelled++;
  // 'boite vocale', 'faux numéro' etc. are negative outcomes too — keep them in cancelled
  if (s === 'boite vocale' || s === 'faux numéro') buckets.cancelled++;
  // 'Pas de réponse' is just "not yet processed" — neither confirmed nor cancelled
  // 'nouveau' is also neither — not counted in any bucket
}Important: delivered + en_cours + (confirme itself representing "currently sitting in 'confirme'") + refused are NOT mutually exclusive with confirme — they're all subsets of it.
