 * Look up Ameex's numeric city ID from a customer-provided city name.
 * Uses normalized matching (lowercase + accent-stripped + whitespace-trimmed).

async function resolveAmeexCityId(storeId: number, cityName: string): Promise<string | null> {
  const norm = (s: string) => s
    .toLowerCase()
    .normalize('NFD').replace(/[\u0300-\u036f]/g, '')   // strip accents
    .replace(/\s+/g, ' ')
    .trim();

  const key = norm(cityName);
  if (!key) return null;

  // Exact match first
  const [exact] = await db.select().from(ameexCities)
    .where(and(eq(ameexCities.storeId, storeId), eq(ameexCities.nameNorm, key)))
    .limit(1);
  if (exact) return exact.externalId;

  // Fuzzy match (LIKE) — returns first hit
  const fuzzy = await db.select().from(ameexCities)
    .where(and(

      sql`${ameexCities.nameNorm} LIKE ${'%' + key + '%'}`
    ))
    .limit(1);
  return fuzzy[0]?.externalId ?? null;
