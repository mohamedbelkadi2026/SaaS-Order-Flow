**Fix 2 — Handle Digylog's specific error format**

Digylog returns errors in this format when something is wrong:
```json
[{ "num": "order-ref", "error": "some error message" }]
```
OR:
```json
{ "message": "...", "errors": {...} }
```

Add a new function `detectDigylogError` BEFORE `detectLogicalError` is called, specifically for Digylog:
```typescript
function detectDigylogError(body: any): string | null {
  if (!body) return null;
  
  // Digylog returns array of order results — check each for errors
  if (Array.isArray(body)) {
    const errors = body.filter((item: any) => item.error || item.errors || item.message);
    if (errors.length > 0) {
      const msg = errors.map((e: any) => e.error || e.message || JSON.stringify(e)).join(", ");
      return msg;
    }
    return null;
  }
  
  // Digylog sometimes wraps in { orders: [...] } with errors inside
  if (Array.isArray(body.orders)) {
    const errors = body.orders.filter((item: any) => item.error || item.errors);
    if (errors.length > 0) {
      return errors.map((e: any) => e.error || JSON.stringify(e)).join(", ");
    }
  }
  
  // Digylog validation errors: { message: "...", errors: { field: ["msg"] } }
  if (body.message && body.errors) {
    const fieldErrors = Object.values(body.errors).flat().join(", ");
    return `${body.message}: ${fieldErrors}`;
  }
  
  return null;
}
```

Then in the response handling section, for Digylog specifically, call this BEFORE `detectLogicalError`:
```typescript
if (providerKey === "digylog") {
  const digylogError = detectDigylogError(rawBody);
  if (digylogError) {
    console.error(`${tag} ❌ Digylog error: ${digylogError}`);
    return {
      success: false,
      httpStatus,
      rawResponse: rawBody,
      error: digylogError,
      carrierMessage: digylogError,
    };
  }
}
```
