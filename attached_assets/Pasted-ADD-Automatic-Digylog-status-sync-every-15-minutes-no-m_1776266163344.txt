typescript// ── Auto Digylog sync every 15 minutes ─────────────────────────────
const autoDigylogSync = setInterval(async () => {
  try {
    const { storage } = await import('./storage');
    const { trackDigylogShipment } = await import('./services/carrier-service');
    
    // Get all stores with active Digylog accounts
    const { db } = await import('./db');
    const { carrierAccounts } = await import('@shared/schema');
    const { eq } = await import('drizzle-orm');
    
    const accounts = await db.select().from(carrierAccounts)
      .where(eq(carrierAccounts.carrierName, 'digylog'));
    
    for (const account of accounts) {
      const storeId = account.storeId;
      const apiKey = account.apiKey;
      
      // Get all in-transit orders for this store
      const allOrders = await storage.getOrdersByStore(storeId);
      const toSync = allOrders.filter((o: any) =>
        o.shippingProvider === 'digylog' &&
        o.trackNumber &&
        !['delivered', 'refused', 'Retour Recu'].includes(o.status || '')
      );
      
      if (toSync.length === 0) continue;
      
      console.log(`[AUTO-SYNC] Digylog store=${storeId}: syncing ${toSync.length} orders`);
      
      for (const order of toSync) {
        const result = await trackDigylogShipment(order.trackNumber!, apiKey);
        if (result.status && result.status !== order.status) {
          await storage.updateOrderStatus(order.id, result.status);
          await storage.updateOrder(order.id, { commentStatus: result.rawStatus || result.status });
          await storage.createOrderFollowUpLog({
            orderId: order.id,
            agentId: null,
            agentName: 'Digylog Auto-Sync',
            note: `📦 Statut mis à jour automatiquement: ${result.rawStatus} → ${result.status}`,
          });
          console.log(`[AUTO-SYNC] Order #${(order as any).orderNumber} → ${result.rawStatus} (${result.status})`);
          
          // Broadcast real-time update
          try {
            const { broadcastToStore } = await import('./sse');
            broadcastToStore(storeId, 'order_updated', {
              orderId: order.id,
              status: result.status,
              commentStatus: result.rawStatus,
            });
          } catch {}
        }
      }
    }
  } catch (err: any) {
    console.error('[AUTO-SYNC] Digylog sync error:', err?.message);
  }
}, 15 * 60 * 1000); // every 15 minutes
