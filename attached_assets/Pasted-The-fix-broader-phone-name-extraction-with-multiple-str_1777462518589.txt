The fix — broader phone+name extraction with multiple strategies
In server/services/carrier-service.ts around line 1170, replace the historics extraction block with this version:
tsconst allRecords = Array.isArray(records) ? records : [];
let histDriverPhone = "";
let histDriverName  = "";

// Strategy: scan every record's comment/note text. Try multiple patterns.
// Stop at the first record that yields a phone — historics are newest-first
// so this gets the most recent driver assignment.
for (const rec of allRecords) {
  const commentText = String(
    rec?.comment || rec?.COMMENT || rec?.note || rec?.newvalue || rec?.location || ""
  );
  if (!commentText) continue;

  // Strategy 1 — explicit "téléphone:" / "tél:" / "tel:" prefix
  let m = commentText.match(/(?:t[ée]l[ée]phone|t[ée]l|phone)\s*[:=\-]?\s*\+?(?:212|0)?([0-9\s.-]{8,12})/i);
  if (m) {
    const cleaned = m[1].replace(/\D/g, '');
    if (cleaned.length >= 8 && cleaned.length <= 10) {
      histDriverPhone = cleaned.length === 9 ? '0' + cleaned : cleaned;
    }
  }

  // Strategy 2 — bare 10-digit Moroccan mobile (06xx or 07xx)
  if (!histDriverPhone) {
    m = commentText.match(/(?:^|[^0-9])(0[67][0-9]{8})(?:[^0-9]|$)/);
    if (m) histDriverPhone = m[1];
  }

  // Strategy 3 — international format +212 6/7 xxxxxxx
  if (!histDriverPhone) {
    m = commentText.match(/\+?212[\s.-]?([67][0-9]{8})/);
    if (m) histDriverPhone = '0' + m[1];
  }

  // Strategy: extract driver name when the comment mentions an assignment.
  // Common patterns: "Affecté à Karim", "Livreur: Hassan", "Assigné à Mohamed",
  //                  "Driver: Karim 0607...", "Hassan - 0607394948"
  if (!histDriverName) {
    const nameM =
      commentText.match(/(?:affect[ée]|assign[ée])\s*(?:à|a)?\s*([A-Za-zÀ-ÿ' .-]{3,40})/i) ||
      commentText.match(/(?:livreur|driver|chauffeur)\s*[:=\-]?\s*([A-Za-zÀ-ÿ' .-]{3,40})/i);
    if (nameM) {
      histDriverName = nameM[1].trim()
        .replace(/[\d+]/g, '')           // strip phone digits if mixed in
        .replace(/[\-:,].*$/, '')        // strip trailing separator + phone
        .trim();
      if (histDriverName.length < 2) histDriverName = "";
    }
  }

  if (histDriverPhone) break;
}
