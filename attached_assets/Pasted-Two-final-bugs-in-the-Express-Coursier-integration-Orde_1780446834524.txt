STEP 1 — Make the resolver NEVER return a non-numeric value:

  async resolveExpressCoursierCityId(cityName: string, accountId?: number): Promise<string | null> {
    const norm = (s: string) => (s || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").trim();
    const target = norm(cityName);
    if (!target) return null;

    // 1) Per-account carrier_cities (preferred)
    if (accountId) {
      const rows = await db.select().from(carrierCities).where(eq(carrierCities.accountId, accountId));
      for (const r of rows) {
        if (norm(r.cityName) === target) {
          const ext =
            (r as any).settings?.cityId ||
            (r as any).externalCityId ||
            (r as any).cityId ||
            null;
          // Only return if it's actually numeric
          if (ext && /^\d+$/.test(String(ext))) return String(ext);
        }
      }
    }

    // 2) Hardcoded fallback (MUST be numeric)
    const fallback: Record<string, string> = {
      "casablanca": "337",
      "rabat": "338",
      "marrakech": "339",
      "tanger": "340",
      "berrechid": "341",
      "agadir": "342",
      "fes": "343",
      "kenitra": "344",
      // ⚠️ These are placeholders. Replace with real Express Coursier city IDs.
    };
    if (fallback[target] && /^\d+$/.test(fallback[target])) return fallback[target];

    // 3) Not found → return null (DO NOT fall back to the name)
    return null;
  }

STEP 3 — Add a manual city seeding endpoint so the user can add Azrou and others NOW:

In server/routes.ts:

  app.post("/api/carriers/expresscoursier/seed-cities/:accountId", requireAuth, async (req, res) => {
    try {
      const accountId = parseInt(req.params.accountId);
      const account = await storage.getCarrierAccount(accountId);
      if (!account || account.carrierName !== "expresscoursier") {
        return res.status(404).json({ message: "Compte Express Coursier introuvable" });
      }
      const cities: Array<{ cityId: string; cityName: string }> = req.body.cities || [];
      if (!Array.isArray(cities) || cities.length === 0) {
        return res.status(400).json({ message: "Liste de villes requise" });
      }
      let inserted = 0;
      for (const c of cities) {
        const cityId = String(c.cityId || "").trim();
        const cityName = String(c.cityName || "").trim();
        if (!/^\d+$/.test(cityId) || !cityName) continue;
        // Upsert: insert if not exists for this account+cityName
        const existing = await db.select().from(carrierCities)
          .where(and(eq(carrierCities.accountId, accountId), eq(carrierCities.cityName, cityName)));
        if (existing.length === 0) {
          await db.insert(carrierCities).values({
            accountId,
            cityName,
            // Store the numeric ID — adapt to actual column or settings JSON
            settings: { cityId },
          } as any);
          inserted += 1;
        } else {
          // Update existing
          await db.update(carrierCities)
            .set({ settings: { ...(existing[0] as any).settings, cityId } } as any)
            .where(eq(carrierCities.id, existing[0].id));
        }
      }
      res.json({ ok: true, inserted, total: cities.length });
    } catch (err: any) {
      console.error("[Seed Cities] error:", err);
      res.status(500).json({ message: err.message || "Erreur" });
    }
  });
