B) Delete a single product:

  app.delete("/api/products/:id", requireAuth, async (req: any, res: any) => {
    const storeId = req.user!.storeId!;
    const productId = parseInt(req.params.id);
    const force = req.query.force === "true";  // allow deletion even if has orders (soft delete)

    try {
      const product = await storage.getProduct(productId);
      if (!product || product.storeId !== storeId) {
        return res.status(404).json({ message: "Produit introuvable" });
      }

      const usage = await storage.getProductUsage(storeId, productId);

      // Has linked orders? Decide based on force flag
      if (usage.ordersCount > 0 && !force) {
        return res.status(409).json({
          message: "Ce produit est lié à des commandes",
          usage,
          hint: "Utilisez force=true pour archiver (soft delete) au lieu de supprimer",
        });
      }

      if (usage.ordersCount > 0 && force) {
        // Soft delete: mark as archived, keep order_items intact
        await storage.archiveProduct(productId);
        return res.json({ ok: true, mode: "archived" });
      }

      // Hard delete (no linked orders)
      await storage.deleteProduct(productId);
      res.json({ ok: true, mode: "deleted" });
    } catch (err: any) {
      console.error("[Delete Product] error:", err);
      res.status(500).json({ message: err.message });
    }
  });

C) Bulk delete:

  app.post("/api/products/bulk-delete", requireAuth, async (req: any, res: any) => {
    const storeId = req.user!.storeId!;
    const { productIds, force } = req.body || {};

    if (!Array.isArray(productIds) || productIds.length === 0) {
      return res.status(400).json({ message: "Liste de produits vide" });
    }
    if (productIds.length > 500) {
      return res.status(400).json({ message: "Maximum 500 produits par opération" });
    }

    const results = { deleted: 0, archived: 0, skipped: 0, errors: [] as any[] };

    for (const id of productIds) {
      try {
        const product = await storage.getProduct(id);
        if (!product || product.storeId !== storeId) {
          results.skipped += 1;
          continue;
        }
        const usage = await storage.getProductUsage(storeId, id);
        if (usage.ordersCount > 0) {
          if (force) {
            await storage.archiveProduct(id);
            results.archived += 1;
          } else {
            results.skipped += 1;
          }
        } else {
          await storage.deleteProduct(id);
          results.deleted += 1;
        }
      } catch (err: any) {
        results.errors.push({ id, message: err.message });
      }
    }

    res.json(results);
  });

D) Smart cleanup filter endpoint:

  app.get("/api/products/cleanup-suggestions", requireAuth, async (req: any, res: any) => {
    const storeId = req.user!.storeId!;
    const type = String(req.query.type || "no_orders");  // no_orders | no_deliveries | duplicates | archived

    try {
      let products: any[] = [];

      if (type === "no_orders") {
        // Products with 0 orders ever
        products = await storage.getProductsWithoutOrders(storeId);
      } else if (type === "no_deliveries") {
        // Products with orders but 0 livrées
        products = await storage.getProductsWithoutDeliveries(storeId);
      } else if (type === "duplicates") {
        // Products sharing the same normalized name (and possibly variant) within store
        products = await storage.getDuplicateProducts(storeId);
      } else if (type === "archived") {
        products = await storage.getArchivedProducts(storeId);
      }

      res.json({ type, count: products.length, products });
    } catch (err: any) {
      res.status(500).json({ message: err.message });
    }
  });

STEP 3 — STORAGE METHODS (server/storage.ts)

Add these methods:

  async getProductUsage(storeId: number, productId: number) {
    const items = await db.select().from(orderItems).where(eq(orderItems.productId, productId));
    if (items.length === 0) return { ordersCount: 0, deliveredCount: 0, inStockOrders: 0, totalRevenue: 0 };

    const orderIds = Array.from(new Set(items.map(i => i.orderId)));
    const ordersList = await db.select().from(orders)
      .where(and(eq(orders.storeId, storeId), inArray(orders.id, orderIds)));

    const deliveredCount = ordersList.filter(o => /livr/i.test(String(o.status || ""))).length;
    const inStockOrders = ordersList.filter(o => !/livr|annul|retour/i.test(String(o.status || ""))).length;
    const totalRevenue = ordersList
      .filter(o => /livr/i.test(String(o.status || "")))
      .reduce((sum, o) => sum + (Number(o.totalPrice) || 0), 0);

    return {
      ordersCount: ordersList.length,
      deliveredCount,
      inStockOrders,
      totalRevenue,
    };
  }

  async deleteProduct(productId: number) {
    // Hard delete. Safe only when no order_items reference it.
    await db.delete(products).where(eq(products.id, productId));
  }

  async archiveProduct(productId: number) {
    // Soft delete. Add `archived_at` column if not present (see STEP 4).
    await db.update(products).set({ archivedAt: new Date() } as any).where(eq(products.id, productId));
  }

  async getProductsWithoutOrders(storeId: number) {
    // Products with no row in order_items
    const allProducts = await db.select().from(products).where(eq(products.storeId, storeId));
    const result: any[] = [];
    for (const p of allProducts) {
      const items = await db.select().from(orderItems).where(eq(orderItems.productId, p.id)).limit(1);
      if (items.length === 0) result.push(p);
    }
    return result;
  }

  async getProductsWithoutDeliveries(storeId: number) {
    const all = await db.select().from(products).where(eq(products.storeId, storeId));
    const result: any[] = [];
    for (const p of all) {
      const items = await db.select().from(orderItems).where(eq(orderItems.productId, p.id));
      if (items.length === 0) continue;  // handled by "no_orders" type
      const orderIds = items.map(i => i.orderId);
      const delivered = await db.select().from(orders).where(and(
        inArray(orders.id, orderIds),
        sql`LOWER(${orders.status}) LIKE '%livr%'`
      )).limit(1);
      if (delivered.length === 0) result.push(p);
    }
    return result;
  }

  async getDuplicateProducts(storeId: number) {
    // Group by normalized name; return products in groups of 2+
    const all = await db.select().from(products).where(eq(products.storeId, storeId));
    const norm = (s: string) => (s || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/\s+/g, " ").trim();
    const groups: Record<string, any[]> = {};
    for (const p of all) {
      const key = norm(p.name);
      (groups[key] ||= []).push(p);
    }
    const dupes: any[] = [];
    for (const [key, items] of Object.entries(groups)) {
      if (items.length >= 2) {
        // Keep the most-used one out of the suggested-deletion list
        // Sort by createdAt desc, suggest the older copies for deletion
        const sorted = [...items].sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
        for (let i = 1; i < sorted.length; i++) dupes.push({ ...sorted[i], duplicateGroup: key });
      }
    }
    return dupes;
  }

  async getArchivedProducts(storeId: number) {
    return await db.select().from(products).where(and(
      eq(products.storeId, storeId),
      sql`${products.archivedAt} IS NOT NULL`
    ));
  }

ALSO update getProducts / getInventoryProducts to EXCLUDE archived by default (add a filter param to include archived if requested):
