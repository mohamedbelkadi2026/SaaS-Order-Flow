FEATURE: "Commandes Négligées" — Unworked returned/refused shipped orders
Goal: Add a new page/section showing orders that were shipped (have a track_number), came back as refused/returned, but no suivi agent has touched them (i.e. last_action_by IS NULL or last_action_at is older than 48h). This helps the admin see which orders are being ignored.

Change 1 — server/routes.ts — New endpoint GET /api/orders/neglected
Add this route near the other /api/orders routes:
tsapp.get("/api/orders/neglected", requireAuth, async (req, res) => {
  try {
    const user = req.user!;
    const storeId = user.storeId!;
    const hoursThreshold = Number(req.query.hours || 48);

    const NEGLECTED_STATUSES = [
      'refused',
      'retourné',
      'Tentative échouée',
      'Retour en cours',
      "Retourné à l'expéditeur",
      'Article retourné',
      'Retour en route',
      'Adresse inconnue',
      'Demande retour',
      'Rappel',
      'Reporté',
      'Injoignable',
      'Pas de réponse 1',
      'Pas de réponse 2',
      'Pas de réponse 3',
      'Pas de réponse 4',
    ];

    const cutoff = new Date(Date.now() - hoursThreshold * 60 * 60 * 1000);

    const allOrders = await storage.getOrdersByStore(storeId);

    const neglected = allOrders.filter(o => {
      // Must have been shipped (has track number)
      if (!(o as any).trackNumber) return false;
      // Must be in a return/refused/unworked status
      if (!NEGLECTED_STATUSES.includes(o.status || '')) return false;
      // No agent touched it, OR last touch was more than `hoursThreshold` hours ago
      const lastAction = (o as any).lastActionAt;
      if (!lastAction) return true;
      return new Date(lastAction) < cutoff;
    });

    // Sort by lastActionAt ascending (oldest neglect first)
    neglected.sort((a, b) => {
      const aTime = (a as any).lastActionAt ? new Date((a as any).lastActionAt).getTime() : 0;
      const bTime = (b as any).lastActionAt ? new Date((b as any).lastActionAt).getTime() : 0;
      return aTime - bTime;
    });

    res.json(neglected);
  } catch (err) {
    throw err;
  }
});
