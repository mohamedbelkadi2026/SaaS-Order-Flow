tsxconst { data: agentMyStats } = useQuery({

    if (agentFilters.dateFrom) {

    } else {
      params.set('dateRange', agentFilters.dateRange);
    }
    const r = await fetch(`/api/agents/my-stats?${params}`, { credentials: 'include' });
    return r.json();
  },

});
