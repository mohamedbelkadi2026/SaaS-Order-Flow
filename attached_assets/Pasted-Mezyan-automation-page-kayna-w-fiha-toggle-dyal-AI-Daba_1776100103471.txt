const [autoSettings, setAutoSettings] = useState({
  aiConfirmation: false,
  recoveryMessages: false,
  marketingAuto: false,
});
const [saving, setSaving] = useState(false);

// Load settings on mount:
useEffect(() => {
  fetch('/api/whatsapp/auto-settings')
    .then(r => r.json())
    .then(d => setAutoSettings(d))
    .catch(() => {});
}, []);

const saveSettings = async (newSettings: typeof autoSettings) => {
  setSaving(true);
  await fetch('/api/whatsapp/auto-settings', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(newSettings),
  });
  setSaving(false);
};

const toggle = (key: keyof typeof autoSettings) => {
  const updated = { ...autoSettings, [key]: !autoSettings[key] };
  setAutoSettings(updated);
  saveSettings(updated);
};

CHANGE 2 — server/routes.ts — add GET + POST /api/whatsapp/auto-settings:
typescript// Store settings in memory (persisted to DB via store settings):
const waAutoSettings: Record<number, { aiConfirmation: boolean; recoveryMessages: boolean; marketingAuto: boolean }> = {};

app.get('/api/whatsapp/auto-settings', requireAuth, (req, res) => {
  const storeId = req.user!.storeId!;
  res.json(waAutoSettings[storeId] || { aiConfirmation: false, recoveryMessages: false, marketingAuto: false });
});

app.post('/api/whatsapp/auto-settings', requireAuth, (req, res) => {
  const storeId = req.user!.storeId!;
  waAutoSettings[storeId] = req.body;
  console.log(`[WA Settings] Store ${storeId}:`, req.body);
  res.json({ success: true });
});

// Export getter for use in other files:
export function getWaAutoSettings(storeId: number) {
  return waAutoSettings[storeId] || { aiConfirmation: false, recoveryMessages: false, marketingAuto: false };
}
