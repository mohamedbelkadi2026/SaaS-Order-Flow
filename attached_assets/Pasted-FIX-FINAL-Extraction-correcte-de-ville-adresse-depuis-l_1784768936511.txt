let customerCity = "";
let customerAddress = "";
const accessToken = decrypt(integration.oauthAccessToken || "");
if (accessToken && payload.id) {
  try {
    const fullOrderResp = await fetch(`https://api.youcan.shop/orders/${payload.id}?include=customer,shipping`, {
      headers: { Authorization: `Bearer ${accessToken}` },
    });
    const fullOrder = await fullOrderResp.json() as any;

    const customer = fullOrder?.customer || {};
    const shippingAddr = fullOrder?.shipping?.address || null; // peut être absent selon le type de commande

    // Ville : privilégier shipping.address.city si présent (commandes avec
    // adresse structurée), sinon customer.city (texte libre côté COD).
    customerCity = shippingAddr?.city || customer.city || "";

    // Adresse : region/location du customer contiennent le texte libre saisi
    // par le client (quartier, repère...) — c'est la meilleure adresse
    // disponible pour ce type de commande YouCan.
    customerAddress = shippingAddr?.address
      || customer.location
      || customer.region
      || "";

    // Si "city" ressemble à un code numérique pur (ex: "66") plutôt qu'à un
    // nom de ville, ce n'est pas exploitable tel quel — dans ce cas, se
    // rabattre sur region/location pour au moins avoir un texte utile.
    if (customerCity && /^\d+$/.test(customerCity.trim())) {
      console.warn(`[YOUCAN-WEBHOOK] city looks like a numeric code ("${customerCity}") for order ${payload.id}, falling back to region/location`);
      customerCity = customer.region || customer.location || "";
    }

    if (!customerCity) console.warn(`[YOUCAN-WEBHOOK] Still no usable city for order ${payload.id}`);
  } catch (fetchErr: any) {
    console.error("[YOUCAN-WEBHOOK] Failed to fetch full order for address:", fetchErr.message);
  }
}
