exists in the Ozon webhook branch, but (1) it only runs when the status CHANGES, so already-
delivered orders (delivered before the feature) never got a fee, and (2) there's no backfill.
Ozon's parcel-info endpoint DOES return the fee (DELIVERED-PRICE / RETURNED-PRICE / REFUSED-PRICE)
even though /parcels/tracking is dead. shippingCost is stored in CENTIMES (UI divides by 100).

PART B — make "Synchroniser Ozon Express" backfill fees for existing orders.
Add a helper and call it from the Ozon sync path:

   async function backfillOzonFees(storeId: number) {
     const acct = (await storage.getCarrierAccounts(storeId, "ozonexpress"))[0];
     const cid = acct?.settings?.ozonExpressCustomerId, key = acct?.apiKey;
     if (!cid || !key) return { checked: 0, updated: 0 };
     // Ozon orders in a final state, with a tracking number, and no fee yet:
     const orders = await storage.getOrdersForFeeBackfill(storeId, "ozonexpress"); // status in (delivered,'Retour Recu',refused) AND trackingNumber present AND (shippingCost is null or 0)
     let updated = 0;
     for (const o of orders) {
       try {
         const FormDataFee = (await import("form-data")).default;
         const fd = new FormDataFee(); fd.append("tracking-number", o.trackingNumber);
         const r = await axios.post(`https://api.ozonexpress.ma/customers/${cid}/${key}/parcel-info`,
           fd, { headers: fd.getHeaders(), timeout: 15000, validateStatus: () => true });
         const infos = r.data?.["PARCEL-INFO"]?.["INFOS"];
         const feeRaw = o.status === "delivered" ? infos?.["DELIVERED-PRICE"]
                      : o.status === "Retour Recu" ? infos?.["RETURNED-PRICE"]
                      : infos?.["REFUSED-PRICE"];
         const fee = Number(feeRaw);
         if (Number.isFinite(fee) && fee > 0) {
           await storage.updateOrder(o.id, { shippingCost: Math.round(fee * 100) });
           updated++;
           console.log(`[OZON-FEE-BACKFILL] #${o.orderNumber} fee=${fee} DH`);
         }
       } catch (e:any) { console.log(`[OZON-FEE-BACKFILL] ${o.trackingNumber} failed: ${e?.message}`); }
     }
     return { checked: orders.length, updated };
   }
