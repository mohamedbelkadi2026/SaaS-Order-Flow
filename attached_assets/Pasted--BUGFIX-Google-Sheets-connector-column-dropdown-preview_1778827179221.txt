Add an autoDetectMapping function that runs after preview loads:tsfunction autoDetectMapping(sampleRows: string[][], columnCount: number): Record<string, number | null> {
  const result: Record<string, number | null> = {
    name: null, phone: null, address: null, city: null,
    product: null, price: null, quantity: null,
    note: null, utmCampaign: null, utmSource: null, productId: null,
  };
  if (!sampleRows || sampleRows.length === 0) return result;

  // For each column, examine its content across all sample rows
  for (let col = 0; col < columnCount; col++) {
    const values = sampleRows
      .map(r => (r?.[col] || '').toString().trim())
      .filter(v => v);
    if (values.length === 0) continue;

    const allText = values.join(' ').toLowerCase();
    const firstValue = values[0];

    // Phone: digits, possibly with + or spaces, 8-15 chars
    if (!result.phone && values.every(v => /^[+\d\s\-()]{8,20}$/.test(v) && v.replace(/\D/g, '').length >= 8)) {
      result.phone = col;
      continue;
    }

    // Price: pure number, possibly with decimal
    if (!result.price && values.every(v => /^\d+([.,]\d+)?$/.test(v.replace(/\s/g, '')))) {
      const numericValues = values.map(v => parseFloat(v.replace(',', '.')));
      // Likely price if values are reasonable (10-100000)
      if (numericValues.every(n => n >= 10 && n <= 100000)) {
        result.price = col;
        continue;
      }
    }

    // Quantity: small integer 1-99
    if (!result.quantity && values.every(v => /^\d{1,2}$/.test(v))) {
      const numericValues = values.map(v => parseInt(v, 10));
      if (numericValues.every(n => n >= 1 && n <= 99)) {
        result.quantity = col;
        continue;
      }
    }

    // Name: 2+ words, mostly letters, no digits
    if (!result.name && values.every(v => /^[\p{L}\s.''-]{2,50}$/u.test(v) && !/^\d/.test(v))) {
      result.name = col;
      continue;
    }

    // City: single word, capitalized, mostly letters
    if (!result.city && values.every(v => /^[\p{L}\s]{2,30}$/u.test(v) && v.split(/\s+/).length <= 3)) {
      result.city = col;
      continue;
    }

    // Address: long text with mixed alphanumeric
    if (!result.address && values.every(v => v.length > 8 && /\d/.test(v) && /\p{L}/u.test(v))) {
      result.address = col;
      continue;
    }

    // UTM: lowercase with underscores/hyphens, no spaces
    if (!result.utmSource && allText.match(/facebook|instagram|google|tiktok|youtube|organic|direct/)) {
      result.utmSource = col;
      continue;
    }

    // Product ID: UUID-like or long hex string
    if (!result.productId && values.every(v => /^[a-f0-9-]{20,}$/i.test(v))) {
      result.productId = col;
      continue;
    }
  }

  return result;
