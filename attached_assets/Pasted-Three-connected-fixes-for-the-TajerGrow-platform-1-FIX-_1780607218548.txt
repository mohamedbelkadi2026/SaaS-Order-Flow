PART C — AUTO-LOAD DEFAULTS IN PROFIT ANALYZER
═══════════════════════════════════════════════════════════════════════

Open client/src/pages/profit-analyzer.tsx. Find Step 2 ("Saisie des coûts") where the user enters per-product costs.

STEP C2 — FETCH PRODUCT DEFAULTS

When the user reaches Step 2 (after product selection), call /api/products to get the store's products and match by name to pre-fill the cost rows:

  useEffect(() => {
    async function loadDefaults() {
      try {
        const resp = await fetch("/api/products", { credentials: "include" });
        const products = await resp.json();

        // Build a name → defaults map
        const defaultsByName = new Map<string, any>();
        for (const p of products || []) {
          const name = (p.name || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").trim();
          const d = (p.settings as any)?.profitDefaults;
          if (name && d) defaultsByName.set(name, d);
        }

        // Pre-fill cost rows
        setCostRows((rows) => rows.map((r) => {
          const key = (r.productName || "").toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").trim();
          const d = defaultsByName.get(key);
          if (!d) return r;
          return {
            ...r,
            prixAchatU: r.prixAchatU || d.coutAchat || 0,
            emballageU: r.emballageU || d.coutEmballage || 0,
            confirmU:   r.confirmU   || d.coutConfirmation || 0,
            // shipping comes from CSV column, not defaults
          };
        }));
      } catch (err) {
        console.warn("[ProfitAnalyzer] Could not load product defaults:", err);
      }
    }
    if (activeStep === 2 && costRows.length > 0) loadDefaults();
  }, [activeStep, costRows.length]);
