typescriptapp.get("/api/agents/my-stats", requireAuth, async (req: any, res: any) => {
  try {
    const user = req.user!;
    const storeId = user.storeId!;
    const agentId = user.id;
    const { dateRange, dateFrom, dateTo, city, product } = req.query as Record<string, string>;

    const now = new Date();
    let cutoff = new Date();
    let endDate: Date = new Date();

    // Date range logic
    if (dateFrom) {
      cutoff = new Date(dateFrom + 'T00:00:00');
      endDate = dateTo ? new Date(dateTo + 'T23:59:59') : new Date();
    } else if (dateRange === 'today') {
      cutoff = new Date(); cutoff.setHours(0, 0, 0, 0);
      endDate = new Date(); endDate.setHours(23, 59, 59, 999);
    } else if (dateRange === 'yesterday') {
      cutoff = new Date(); cutoff.setDate(cutoff.getDate() - 1); cutoff.setHours(0, 0, 0, 0);
      endDate = new Date(); endDate.setDate(endDate.getDate() - 1); endDate.setHours(23, 59, 59, 999);
    } else if (dateRange === '7days') {
      cutoff = new Date(); cutoff.setDate(cutoff.getDate() - 6); cutoff.setHours(0, 0, 0, 0);
    } else if (dateRange === 'month') {
      cutoff = new Date(now.getFullYear(), now.getMonth(), 1);
    } else if (dateRange === 'lastmonth') {
      cutoff = new Date(now.getFullYear(), now.getMonth() - 1, 1);
      endDate = new Date(now.getFullYear(), now.getMonth(), 0, 23, 59, 59);
    } else if (dateRange === 'all') {
      cutoff = new Date('2020-01-01');
    } else {
      // default: ce mois
      cutoff = new Date(now.getFullYear(), now.getMonth(), 1);
    }

    const { orders: ordersTable } = await import("@shared/schema");
    const { and, gte, lte, eq: drEq } = await import("drizzle-orm");

    let agentOrders = await db
      .select()
      .from(ordersTable)
      .where(and(
        drEq(ordersTable.storeId, storeId),
        drEq(ordersTable.assignedToId, agentId),
        gte(ordersTable.createdAt, cutoff),
        lte(ordersTable.createdAt, endDate),
      ));

    // Apply city filter
    if (city && city !== 'all') {
      agentOrders = agentOrders.filter(o => 
        (o.customerCity || '').toLowerCase().includes(city.toLowerCase())
      );
    }

    // Apply product filter
    if (product && product !== 'all') {
      agentOrders = agentOrders.filter((o: any) => {
        const name = o.rawProductName || o.items?.[0]?.rawProductName || '';
        return name === product;
      });
    }

    // Stats
    const total = agentOrders.length;
    let confirme = 0, delivered = 0, cancelled = 0, refused = 0, en_cours = 0, nouveau = 0;

    for (const o of agentOrders) {
      const s = (o.status ?? '').toLowerCase().trim();
      if (s === 'nouveau') nouveau++;
      else if (s === 'delivered' || s === 'livré' || s === 'livrée' || s.includes('livrée')) delivered++;
      else if (s === 'confirme' || s === 'confirmé') confirme++;
      else if (s === 'in_progress' || s.includes('livraison') || s.includes('transit')) en_cours++;
      else if (s.includes('annul') || s.includes('refus') || s === 'refused' || s === 'boite vocale') cancelled++;
    }

    const confirmRate = total > 0 ? Math.round(((confirme + delivered + en_cours) / total) * 100) : 0;
    const deliverRate = (confirme + delivered) > 0 ? Math.round((delivered / (confirme + delivered)) * 100) : 0;

    // Daily chart
    const dayMap = new Map<string, { total: number; confirmed: number }>();
    const diffDays = Math.min(Math.ceil((endDate.getTime() - cutoff.getTime()) / (1000 * 60 * 60 * 24)), 30);
    for (let i = diffDays; i >= 0; i--) {
      const d = new Date(endDate);
      d.setDate(d.getDate() - i);
      const key = `${String(d.getDate()).padStart(2, '0')}/${String(d.getMonth() + 1).padStart(2, '0')}`;
      dayMap.set(key, { total: 0, confirmed: 0 });
    }
    for (const o of agentOrders) {
      if (!o.createdAt) continue;
      const d = new Date(o.createdAt);
      const key = `${String(d.getDate()).padStart(2, '0')}/${String(d.getMonth() + 1).padStart(2, '0')}`;
      if (dayMap.has(key)) {
        const curr = dayMap.get(key)!;
        curr.total++;
        const s = (o.status ?? '').toLowerCase();
        if (s === 'confirme' || s === 'confirmé' || s === 'delivered') curr.confirmed++;
        dayMap.set(key, curr);
      }
    }
    const daily = Array.from(dayMap.entries()).map(([date, v]) => ({ date, orders: v.total, confirmed: v.confirmed }));

    // Products
    const productMap: Record<string, { name: string; total: number; confirmed: number; delivered: number }> = {};
    for (const o of agentOrders as any[]) {
      const name = o.rawProductName || o.items?.[0]?.rawProductName || o.items?.[0]?.product?.name || 'Produit';
      if (!productMap[name]) productMap[name] = { name, total: 0, confirmed: 0, delivered: 0 };
      productMap[name].total++;
      const s = (o.status ?? '').toLowerCase();
      if (s === 'confirme' || s === 'confirmé' || s === 'delivered') productMap[name].confirmed++;
      if (s === 'delivered') productMap[name].delivered++;
    }
    const products = Object.values(productMap).sort((a, b) => b.total - a.total);

    // Cities (from actual orders)
    const citiesFromOrders = [...new Set(agentOrders.map(o => o.customerCity).filter(Boolean))];

    res.json({
      total, confirme, delivered, cancelled, refused, en_cours, nouveau,
      confirmRate, deliverRate,
      daily,
      products,
      cities: citiesFromOrders,
    });

  } catch (err: any) {
    console.error("[/api/agents/my-stats]", err.message);
    res.status(500).json({ message: err.message });
  }
});
