ISSUE 1: Ozon Express "Synchroniser les villes" returns 422
═══════════════════════════════════════════════════════════════════════
Error: "422: Aucune ville reçue de ozonexpress. Vérifiez votre token et réessayez."

ROOT CAUSE most likely: the sync endpoint is hitting the wrong URL. The earlier integration assumed `GET https://api.ozonexpress.ma/cities` is a public endpoint, but Ozon Express uses authenticated endpoints scoped per customer:

  Correct pattern:
    GET  https://api.ozonexpress.ma/customers/{CUSTOMER_ID}/{API_KEY}/cities
  Possible alternates (try in order if 404/401):
    POST https://api.ozonexpress.ma/customers/{CUSTOMER_ID}/{API_KEY}/cities
    GET  https://api.ozonexpress.ma/cities/{CUSTOMER_ID}/{API_KEY}
    GET  https://api.ozonexpress.ma/cities (public, may exist)

STEP A — Inspect server/routes.ts → /api/carriers/ozonexpress/sync-cities/:accountId.
Report the exact fetch URL the code uses today.

STEP B — Fix the endpoint to try the authenticated URL FIRST, then fall back:

  app.post("/api/carriers/ozonexpress/sync-cities/:accountId", requireAuth, async (req: any, res: any) => {
    try {
      const accountId = parseInt(req.params.accountId);
      const account = await storage.getCarrierAccount(accountId);
      if (!account || account.carrierName !== "ozonexpress") {
        return res.status(404).json({ message: "Compte Ozon Express introuvable" });
      }

      const settings = (account.settings as any) || {};
      const customerId = String(settings.ozonCustomerId || "").trim();
      const apiKey = String(account.apiKey || "").trim();

      if (!customerId || !apiKey) {
        return res.status(400).json({
          message: "Customer ID ou API Key manquant. Modifiez le compte pour les renseigner."
        });
      }

      // Try authenticated endpoints in order
      const candidateUrls = [
        `https://api.ozonexpress.ma/customers/${encodeURIComponent(customerId)}/${encodeURIComponent(apiKey)}/cities`,
        `https://api.ozonexpress.ma/cities/${encodeURIComponent(customerId)}/${encodeURIComponent(apiKey)}`,
        `https://api.ozonexpress.ma/cities`,
      ];

      let lastError: any = null;
      let cities: any[] = [];
      let usedUrl = "";

      for (const url of candidateUrls) {
        try {
          console.log(`[Ozon SyncCities] Trying ${url}`);
          const resp = await fetch(url, {
            method: "GET",
            headers: { "Accept": "application/json" },
          });
          const ctype = resp.headers.get("content-type") || "";
          const text = await resp.text();
          console.log(`[Ozon SyncCities] ${url} → ${resp.status} (${ctype}) — first 200 chars: ${text.slice(0, 200)}`);

          if (!resp.ok) {
            lastError = `HTTP ${resp.status} on ${url}: ${text.slice(0, 200)}`;
            continue;
          }
          let data: any = null;
          try { data = JSON.parse(text); } catch { lastError = `Non-JSON response from ${url}`; continue; }

          // Adapt to various known shapes
          const candidates =
            (Array.isArray(data) && data) ||
            (Array.isArray(data?.cities) && data.cities) ||
            (Array.isArray(data?.data) && data.data) ||
            (Array.isArray(data?.data?.cities) && data.data.cities) ||
            (data?.cities && typeof data.cities === "object" ? Object.entries(data.cities).map(([id, name]) => ({ id, name })) : null) ||
            [];

          if (candidates.length > 0) {
            cities = candidates;
            usedUrl = url;
            break;
          }
          lastError = `Empty cities array from ${url}`;
        } catch (e: any) {
          lastError = `Fetch error on ${url}: ${e.message}`;
        }
      }

      if (cities.length === 0) {
        return res.status(422).json({
          message: `Aucune ville reçue de ozonexpress. Dernière erreur: ${lastError || "inconnu"}. Vérifiez votre Customer ID et API Key dans Modifier le compte.`,
        });
      }

      let inserted = 0, updated = 0, skipped = 0;
      for (const c of cities) {
        const cityId = String(c.id ?? c.city_id ?? c.ID ?? c.value ?? "").trim();
        const cityName = String(c.name ?? c.city_name ?? c.ville ?? c.label ?? "").trim();
        if (!/^\d+$/.test(cityId) || !cityName) { skipped += 1; continue; }

        const existing = await db.select().from(carrierCities)
          .where(and(eq(carrierCities.accountId, accountId), eq(carrierCities.cityName, cityName)));

        if (existing.length === 0) {
          await db.insert(carrierCities).values({
            accountId, cityName, settings: { cityId },
          } as any);
          inserted += 1;
        } else {
          await db.update(carrierCities)
            .set({ settings: { ...(existing[0] as any).settings, cityId } } as any)
            .where(eq(carrierCities.id, existing[0].id));
          updated += 1;
        }
      }

      res.json({
        ok: true,
        usedUrl,
        inserted,
        updated,
        skipped,
        total: cities.length,
        message: `${inserted} villes ajoutées, ${updated} mises à jour, ${skipped} ignorées`,
      });
    } catch (err: any) {
      console.error("[Ozon SyncCities] error:", err);
      res.status(500).json({ message: err.message || "Erreur" });
    }
  });
