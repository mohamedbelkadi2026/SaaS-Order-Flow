typescriptapp.get("/api/agents/my-stats", requireAuth, async (req: any, res: any) => {
  try {
    const user = req.user!;
    const storeId = user.storeId!;
    const agentId = user.id;

    // NEW: Accept filter params
    const { city, productId, dateFrom, dateTo } = req.query as Record<string, string>;

    const now = new Date();
    const cutoff = new Date(now);
    cutoff.setDate(cutoff.getDate() - 14);
    cutoff.setHours(0, 0, 0, 0);

    const { orders: ordersTable } = await import("@shared/schema");
    const { and, gte, eq: drEq } = await import("drizzle-orm");

    let agentOrders = await db
      .select()
      .from(ordersTable)
      .where(and(
        drEq(ordersTable.storeId, storeId),
        drEq(ordersTable.assignedToId, agentId),
        gte(ordersTable.createdAt, dateFrom ? new Date(dateFrom) : cutoff),
      ));

    // Apply filters
    if (city && city !== 'all') {
      agentOrders = agentOrders.filter(o => o.customerCity === city);
    }
    if (productId && productId !== 'all') {
      const pid = Number(productId);
      agentOrders = agentOrders.filter(o => 
        (o as any).items?.some((i: any) => i.productId === pid)
      );
    }
    if (dateTo) {
      const to = new Date(dateTo);
      to.setHours(23, 59, 59, 999);
      agentOrders = agentOrders.filter(o => o.createdAt && new Date(o.createdAt) <= to);
    }

    // Build unique cities and products for filter options
    const cities = [...new Set(agentOrders.map(o => o.customerCity).filter(Boolean))];
    const productMap: Record<string, { id: number; name: string; total: number; confirmed: number; delivered: number }> = {};
    agentOrders.forEach((o: any) => {
      const name = o.rawProductName || o.items?.[0]?.rawProductName || o.items?.[0]?.product?.name || 'Produit';
      const pid = o.items?.[0]?.productId || 0;
      if (!productMap[name]) productMap[name] = { id: pid, name, total: 0, confirmed: 0, delivered: 0 };
      productMap[name].total++;
      if (['confirme','delivered','in_progress','expédié'].includes(o.status || '')) productMap[name].confirmed++;
      if (o.status === 'delivered') productMap[name].delivered++;
    });
    const products = Object.values(productMap).sort((a, b) => b.total - a.total);

    // ... rest of existing stats calculation unchanged ...

    res.json({ daily, byStatus, totalOrders: agentOrders.length, cities, products });
  }
